	(
	rut varchar(10) not null PRIMARY KEY,
	nom_ape varchar (100) not null,	
	alias varchar(40) not null unique,
	email varchar (60) not null,
	region_id numeric (2) not null,	
	comuna_id numeric (4) not null,
	candidato_id numeric (3) not null,
	publicidad varchar (100) not null,
	FOREIGN KEY (region_id) REFERENCES region (region_id),
	FOREIGN KEY (comuna_id) REFERENCES comuna (comuna_id),
	FOREIGN KEY (candidato_id) REFERENCES candidato (candidato_id)
